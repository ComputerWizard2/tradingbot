
   ```python
   from models.dreamer_agent import DreamerV3Agent
   from models.mcts import DreamerMCTSAgent
   from models.risk_supervisor import SafeTradingAgent
   from models.ensemble import EnsembleAgent

   # Load trained agent
   agent = DreamerV3Agent(obs_dim=705, action_dim=2)
   agent.load("train/dreamer/ppo_xauusd_latest.pt")

   # Add MCTS planning
   mcts_agent = DreamerMCTSAgent(agent, num_simulations=50)

   # Create ensemble (train 5 versions)
   ensemble = EnsembleAgent(agent_class=DreamerV3Agent, num_models=5)

   # Wrap with safety
   safe_agent = SafeTradingAgent(ensemble, risk_supervisor)

   # Trade!
   action, info = safe_agent.act(obs, state, market_data)
   ```
